function fadeOutAudio(audioEl, duration = 1200){

<script>
(function(){
/* ---------- Canvas bubbles (neon) ---------- */
const canvas = document.getElementById('faqCanvas');
const ctx = canvas.getContext('2d', { alpha: true });
let W = canvas.width = window.innerWidth;
let H = canvas.height = window.innerHeight;
const bubbles = [];
const COLORS = ['rgba(168,85,247,0.14)', 'rgba(0,229,255,0.12)', 'rgba(168,85,247,0.08)'];

function rand(min,max){ return Math.random()*(max-min)+min; }
function createBubbles(){
bubbles.length = 0;
const count = Math.max(18, Math.floor((W*H)/90000));
for(let i=0;i<count;i++){
bubbles.push({
x: rand(0,W),
y: rand(0,H),
r: rand(18,90),
vx: rand(-0.25,0.25),
vy: rand(-0.1, -0.4),
col: COLORS[Math.floor(Math.random()*COLORS.length)],
blur: rand(6,30)
});
}
}
function draw(){
ctx.clearRect(0,0,W,H);
for(const b of bubbles){
ctx.beginPath();
ctx.fillStyle = b.col;
ctx.shadowBlur = b.blur;
ctx.shadowColor = b.col;
ctx.arc(b.x,b.y,b.r,0,Math.PI*2);
ctx.fill();
b.x += b.vx;
b.y += b.vy;
if(b.y + b.r < -60) { b.y = H + b.r + rand(0,120); b.x = rand(0,W); }
if(b.x < -200) b.x = W + 200;
if(b.x > W + 200) b.x = -200;
}
requestAnimationFrame(draw);
}
window.addEventListener('resize', ()=>{
W = canvas.width = window.innerWidth;
H = canvas.height = window.innerHeight;
createBubbles();
});
createBubbles(); draw();

/* ---------- FAQ logic ---------- */
const faqList = document.getElementById('faqList');
const search = document.getElementById('faqSearch');
const tabs = document.querySelectorAll('.tab');
const expandAllBtn = document.getElementById('expandAll');
const collapseAllBtn = document.getElementById('collapseAll');
const likePrefix = 'faq_likes_v2';
const dislikePrefix = 'faq_dislikes_v2';
const openKey = 'faq_last_open_v2';

// load counts from localStorage (or init)
function loadCounts() {
let likes = JSON.parse(localStorage.getItem(likePrefix) || '{}');
let dislikes = JSON.parse(localStorage.getItem(dislikePrefix) || '{}');
document.querySelectorAll('.like').forEach(btn=>{
const id = btn.dataset.id;
const span = document.getElementById('like-'+id);
span.textContent = likes[id] || 0;
});
document.querySelectorAll('.dislike').forEach(btn=>{
const id = btn.dataset.id;
const span = document.getElementById('dislike-'+id);
span.textContent = dislikes[id] || 0;
});
}
function saveCount(type, id, value){
const key = (type === 'like' ? likePrefix : dislikePrefix);
const obj = JSON.parse(localStorage.getItem(key) || '{}');
obj[id] = value;
localStorage.setItem(key, JSON.stringify(obj));
}

// register like/dislike handlers
document.querySelectorAll('.like').forEach(btn=>{
btn.addEventListener('click', (e)=>{
const id = btn.dataset.id;
const span = document.getElementById('like-'+id);
let v = parseInt(span.textContent || '0') + 1;
span.textContent = v;
saveCount('like', id, v);
// mild feedback: brief pulse
btn.animate([{ transform:'scale(1)' }, { transform:'scale(1.06)' }, { transform:'scale(1)' }], { duration:250 });
});
});
document.querySelectorAll('.dislike').forEach(btn=>{
btn.addEventListener('click', (e)=>{
const id = btn.dataset.id;
const span = document.getElementById('dislike-'+id);
let v = parseInt(span.textContent || '0') + 1;
span.textContent = v;
saveCount('dislike', id, v);
btn.animate([{ transform:'scale(1)' }, { transform:'scale(1.06)' }, { transform:'scale(1)' }], { duration:250 });
});
});

loadCounts();

// accordion toggle
function closeItem(item){
const btn = item.querySelector('.faq-toggle');
const ans = item.querySelector('.faq-a');
btn.setAttribute('aria-expanded','false');
ans.hidden = true;
ans.classList.remove('show');
btn.querySelector('.chev').style.transform = '';
}
function openItem(item){
const btn = item.querySelector('.faq-toggle');
const ans = item.querySelector('.faq-a');
btn.setAttribute('aria-expanded','true');
ans.hidden = false;
ans.classList.add('show');
btn.querySelector('.chev').style.transform = 'rotate(180deg)';
// save last opened
const id = item.dataset.id;
localStorage.setItem(openKey, id);
// scroll into view smoothly
setTimeout(()=> item.scrollIntoView({ behavior:'smooth', block: 'center' }), 140);
}

// attach toggles
document.querySelectorAll('.faq-item').forEach(item=>{
const btn = item.querySelector('.faq-toggle');
btn.addEventListener('click', ()=>{
const isOpen = btn.getAttribute('aria-expanded') === 'true';
if(isOpen){ closeItem(item); localStorage.removeItem(openKey); }
else { openItem(item); }
});
});

// Expand / Collapse all
expandAllBtn.addEventListener('click', ()=> {
document.querySelectorAll('.faq-item').forEach(item=> openItem(item));
});
collapseAllBtn.addEventListener('click', ()=> {
document.querySelectorAll('.faq-item').forEach(item=> closeItem(item));
localStorage.removeItem(openKey);
});

// Restore last opened
const last = localStorage.getItem(openKey);
if(last){
const item = document.querySelector('.faq-item[data-id="'+last+'"]');
if(item) openItem(item);
}

// category filter
tabs.forEach(t => {
t.addEventListener('click', ()=>{
tabs.forEach(x=> x.classList.remove('active'));
t.classList.add('active');
const cat = t.dataset.cat;
document.querySelectorAll('.faq-item').forEach(item=>{
if(cat === 'all' || item.dataset.cat === cat) item.style.display = '';
else item.style.display = 'none';
});
});
});

// search
search.addEventListener('input', ()=> {
const q = search.value.trim().toLowerCase();
document.querySelectorAll('.faq-item').forEach(item=>{
const txt = (item.innerText || '').toLowerCase();
item.style.display = txt.includes(q) ? '' : 'none';
});
});

// feedback form (very simple)
const form = document.getElementById('faqForm');
const msg = document.getElementById('faqFormMsg');
form.addEventListener('submit', (e)=>{
e.preventDefault();
const input = document.getElementById('userQ');
const val = input.value.trim();
if(!val){ msg.textContent = 'Please write your question first.'; msg.style.color='#ffd6d6'; return; }
// here you can replace with real submission (AJAX / fetch)
input.value = '';
msg.style.color = '#c6ffd6';
msg.textContent = 'Thanks — your question has been received. We will reply soon!';
setTimeout(()=> msg.textContent = '', 4500);
});

// ensure colors and visibility (quick safeguard)
document.querySelectorAll('.faq-title, .faq-toggle, .faq-a, .tab, .small-btn').forEach(n=>{
if(getComputedStyle(n).color === 'rgb(0, 0, 0)') n.style.color = '#fff';
});

})();
